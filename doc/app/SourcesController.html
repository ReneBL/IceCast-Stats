def set_source
        source = params[:source]
        unless (!source.eql? "Todos")
                session[:source] = nil
        end
        if (sources_file_lines.include? source)
                session[:source] = source
                render :json => {"source" => source}.to_json
        else
                render :json => {"source" => "does not exist"}.to_json
        end
end
